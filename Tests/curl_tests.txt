### Graph Services ###

## About
curl -X POST http://devapi.opentreeoflife.org/v2/graph/about

## Get source tree from the graph
curl -X POST http://devapi.opentreeoflife.org/v2/graph/source_tree -H "content-type:application/json" -d '{"tree_id":"pg_420_522_96e3dcc7d18b5ba7b96b888ef18fdf7c14c088fa"}'

## Node info
curl -X POST http://devapi.opentreeoflife.org/v2/graph/node_info -H "content-type:application/json" -d '{"ott_id":292466}'

curl -X POST http://devapi.opentreeoflife.org/v2/graph/node_info -H "content-type:application/json" -d '{"node_id":3019509}'

# Aves: ott_id = 81461; node_id = 3019459

# Passeroidea: in graph, but not synth
curl -X POST http://devapi.opentreeoflife.org/v2/graph/node_info -H "content-type:application/json" -d '{"ott_id":176458}'

### Tree Services ###

## About
curl -X POST http://devapi.opentreeoflife.org/v2/tree_of_life/about

## Get MRCA of nodes
curl -X POST http://devapi.opentreeoflife.org/v2/tree_of_life/mrca -H "content-type:application/json" -d '{"ott_ids":[292466, 501678, 267845]}'

curl -X POST http://devapi.opentreeoflife.org/v2/tree_of_life/mrca -H "content-type:application/json" -d '{"ott_ids":[292466]}'

# Example where MRCA node does not have a name
curl -X POST http://devapi.opentreeoflife.org/v2/tree_of_life/mrca -H "content-type:application/json" -d '{"ott_ids":[412129, 536234]}'

## Get subtree below some node
# Parulidae
curl -X POST http://devapi.opentreeoflife.org/v2/tree_of_life/subtree -H "content-type:application/json" -d '{"ott_id":3599390}'

# Passeroidea (not in synth tree)
curl -X POST http://devapi.opentreeoflife.org/v2/tree_of_life/subtree -H "content-type:application/json" -d '{"ott_id":176458}'

# Bad call: provides both ott_id and node_id
curl -X POST http://devapi.opentreeoflife.org/v2/tree_of_life/subtree -H "content-type:application/json" -d '{"ott_id":3599390, "node_id":3599390}'

# Get complete Aves tree
curl -X POST http://devapi.opentreeoflife.org/v2/tree_of_life/subtree -H "content-type:application/json" -d '{"ott_id":81461}'

curl -X POST http://devapi.opentreeoflife.org/v2/tree_of_life/subtree -H "content-type:application/json" -d '{"ott_id": 801601}'

## Get induced subtree = synth tree with just specified terminal nodes
curl -X POST http://devapi.opentreeoflife.org/v2/tree_of_life/induced_subtree -H "content-type:application/json" -d '{"ott_ids":[292466, 501678, 267845, 666104, 316878, 102710]}'
